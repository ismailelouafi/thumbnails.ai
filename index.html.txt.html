        /**

        function buildEnhancedPrompt(basePrompt, options) {

        /**
         * Convert aspect ratio to image size string (for DALL-E)
         */
        function getImageSize(aspectRatio) {
            const sizeMap = {
                '16:9': '1792x1024',
                '9:16': '1024x1792',
                '1:1': '1024x1024',
                '4:5': '1024x1280'
            };
            return sizeMap[aspectRatio] || '1024x1024';
        }

        /**
         * Convert aspect ratio to dimensions object
         */
        function getImageDimensions(aspectRatio) {
            const dimensionMap = {
                '16:9': { width: 1280, height: 720 },
                '9:16': { width: 720, height: 1280 },
                '1:1': { width: 1024, height: 1024 },
                '4:5': { width: 1024, height: 1280 }
            };
            return dimensionMap[aspectRatio] || { width: 1024, height: 1024 };
        }

        window.addEventListener('DOMContentLoaded', () => {

            if (savedApiKey) {
                elements.apiKeyInput.value = savedApiKey;
            }

        // Save API key to localStorage when changed
        elements.apiKeyInput.addEventListener('change', (e) => {
            const apiKey = e.target.value.trim();
            if (apiKey) {
                localStorage.setItem('wafi_api_key', apiKey);
            } else {
                localStorage.removeItem('wafi_api_key');
            }

        function showLoadingState() {
            elements.emptyState.classList.add('hidden');
            elements.resultState.classList.add('hidden');
            elements.loadingState.classList.remove('hidden');
            elements.generateBtn.disabled = true;
        }

        function showResultState(imageUrl) {
            elements.emptyState.classList.add('hidden');
            elements.loadingState.classList.add('hidden');
            elements.resultState.classList.remove('hidden');
            elements.resultImage.src = imageUrl;
            elements.generateBtn.disabled = false;
            currentImageUrl = imageUrl;
        }

        function showEmptyState() {
            elements.emptyState.classList.remove('hidden');
            elements.loadingState.classList.add('hidden');
            elements.resultState.classList.add('hidden');
            elements.generateBtn.disabled = false;
        }

            alert(`Error: ${message}`);

        async function generateThumbnail() {
            try {
                // Get API key from localStorage
                const apiKey = localStorage.getItem('wafi_api_key');

                // Collect form data
                const formData = {
                    prompt: elements.promptInput.value.trim(),
                    baseImage: elements.baseImageInput.files[0] || null,
                    options: {
                        designStyle: elements.designStyleSelect.value,
                        emotion: elements.emotionSelect.value,
                        theme: elements.themeSelect.value,
                        aspectRatio: elements.aspectRatioSelect.value
                    }
                };

                // Store for regeneration
                lastFormData = formData;

                // Show loading state
                showLoadingState();

                // Call API
                const imageUrl = await callAiImageApi(
                    formData.prompt,
                    formData.baseImage,
                    formData.options,
                    apiKey
                );

                // Show result
                showResultState(imageUrl);

            } catch (error) {
                console.error('Generation error:', error);
                showError(error.message || 'Failed to generate thumbnail. Please check your API key and try again.');
            }
        }

        elements.downloadBtn.addEventListener('click', async () => {
            if (!currentImageUrl) return;

            try {
                const response = await fetch(currentImageUrl);
                const blob = await response.blob();
                const url = window.URL.createObjectURL(blob);
                const a = document.createElement('a');
                a.href = url;
                a.download = `wafi-thumbnail-${Date.now()}.png`;
                document.body.appendChild(a);
                a.click();
                document.body.removeChild(a);
                window.URL.revokeObjectURL(url);
            } catch (error) {
                console.error('Download error:', error);
                alert('Failed to download image. Please try right-clicking and saving the image manually.');
            }
        });
